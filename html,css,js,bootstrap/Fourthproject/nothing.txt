    function topFunction() {
    document.body.scrollTop = 0;

  }





  function checkSlideIn() {
    const elements = document.querySelectorAll(".slide-in-bottom");
    const scrollPos = window.innerHeight + window.scrollY;

    elements.forEach((element) => {
        if (element.offsetTop < scrollPos) {
            element.classList.add("active");
        }
    });
}


document.addEventListener("scroll", checkSlideIn);
document.addEventListener("DOMContentLoaded", checkSlideIn);
